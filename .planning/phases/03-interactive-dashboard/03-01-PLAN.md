  // ── Column preset apply ───────────────────────────────────────────────────
  function applyPreset(table, preset, allFields) {
    allFields.forEach(function(field) {
      if (preset === "summary") {
        SUMMARY_COLS.has(field) ? table.showColumn(field) : table.hideColumn(field);
      } else {
        table.showColumn(field);
      }
    });
  }

  // ── Main: fetch data and build table ──────────────────────────────────────
  // Cache-bust per CLAUDE.md gotcha: ?v= Date.now()
  fetch("data/results.json?v=" + Date.now())
    .then(function(r) {
      if (!r.ok) throw new Error("HTTP " + r.status + " fetching results.json");
      return r.json();
    })
    .then(function(data) {
      updateFreshnessUI(data.generated_at);

      // Derive full field list from first row (all 43 columns; order matches buildColumns)
      const allFields = buildColumns().map(function(c) { return c.field; });

      var table = new Tabulator("#data-table", {
        data: data.rows,
        layout: "fitData",          // "fitDataFill" is bugged with frozen columns — RESEARCH pitfall 1
        height: "calc(100vh - 140px)",
        renderVertical: "virtual",  // safe; do NOT set renderHorizontal (pitfall 2)
        initialSort: [{ column: "CombinedScore", dir: "desc" }],
        columns: buildColumns(),
      });

      table.on("tableBuilt", function() {
        // Default filters on load
        table.setFilter(noErrorFilter);   // hide error rows (FE-07)

        // Default preset: Summary (D-03)
        applyPreset(table, "summary", allFields);

        // ── Buy Signals toggle (FE-11) ────────────────────────────────────
        // Show column is a JSON boolean true/false (NOT string "True"/"False") — RESEARCH pitfall 4
        var buyActive = false;
        var btnBuy = document.getElementById("btn-buy");
        btnBuy.addEventListener("click", function() {
          buyActive = !buyActive;
          btnBuy.classList.toggle("active", buyActive);
          btnBuy.setAttribute("aria-pressed", String(buyActive));
          if (buyActive) {
            table.addFilter("Show", "=", true);
          } else {
            table.removeFilter("Show", "=", true);
          }
        });

        // ── Error rows toggle (FE-07) ─────────────────────────────────────
        // Uses function reference — same ref required for removeFilter (pitfall 3)
        var errorsVisible = false;
        var btnErrors = document.getElementById("btn-errors");
        btnErrors.addEventListener("click", function() {
          errorsVisible = !errorsVisible;
          btnErrors.classList.toggle("active", errorsVisible);
          btnErrors.setAttribute("aria-pressed", String(errorsVisible));
          if (errorsVisible) {
            table.removeFilter(noErrorFilter);
          } else {
            table.setFilter(noErrorFilter);
          }
        });

        // ── Summary / Full preset toggle (FE-13) ─────────────────────────
        var currentPreset = "summary";
        var btnSummary = document.getElementById("btn-summary");
        var btnFull    = document.getElementById("btn-full");

        btnSummary.addEventListener("click", function() {
          if (currentPreset === "summary") return;
          currentPreset = "summary";
          applyPreset(table, "summary", allFields);
          btnSummary.classList.add("active");    btnSummary.setAttribute("aria-pressed", "true");
          btnFull.classList.remove("active");    btnFull.setAttribute("aria-pressed", "false");
        });

        btnFull.addEventListener("click", function() {
          if (currentPreset === "full") return;
          currentPreset = "full";
          applyPreset(table, "full", allFields);
          btnFull.classList.add("active");       btnFull.setAttribute("aria-pressed", "true");
          btnSummary.classList.remove("active"); btnSummary.setAttribute("aria-pressed", "false");
        });

        // ── Ticker search (FE-14) ─────────────────────────────────────────
        var searchInput = document.getElementById("ticker-search");
        var searchTimeout;
        searchInput.addEventListener("input", function() {
          clearTimeout(searchTimeout);
          var val = this.value.trim().toUpperCase();
          searchTimeout = setTimeout(function() {
            if (val) {
              table.setFilter("Ticker", "like", val);
            } else {
              table.removeFilter("Ticker", "like", "");
            }
          }, 120);
        });
      });
    })
    .catch(function(err) {
      document.getElementById("freshness-badge").textContent = "Error loading data";
      console.error("Dashboard load error:", err);
    });
  </script>

</body>
</html>
